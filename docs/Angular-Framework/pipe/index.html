Pipe

{{today | date:'yyyy/MM/dd HH:mm:ss'}}

當然，現在有很多套件其實也都可以很輕鬆地辦到，但其實 Angular 除了 DatePipe 之外，很貼心地提供了非常多類似的 Pipe 如：

  • 處理貨幣單位非常方便的 CurrencyPipe
  • 自動 JSON 格式化的 JsonPipe
  • 幫你自動訂閱的 AsyncPipe
  • 英文大小寫處理的 UpperCasePipe、LowerCasePipe 與 TitleCasePipe
  • 專門處理小數的 DecimalPipe
  • 小數點轉百分比很方便的 PercentPipe
  • 陣列跟字串都可以用的 SlicePipe
  • 多國語系處理專門的 I18nSelectPipe 與 I18nPluralPipe
  • 暫時想不到用途的 KeyValuePipe

下列是客製化 pipe 的用法

import { Pipe, PipeTransform } from '@angular/core';

/*
 * Raise the value exponentially
 * Takes an exponent argument that defaults to 1.
 * Usage:
 *   value | exponentialStrength:exponent
 * Example:
 *   {{ 2 | exponentialStrength:10 }}
 *   formats to: 1024
*/
@Pipe({name: 'exponentialStrength'})
export class ExponentialStrengthPipe implements PipeTransform {
  transform(value: number, exponent: string): number {
    let exp = parseFloat(exponent);
    return Math.pow(value, isNaN(exp) ? 1 : exp);
  }
}
